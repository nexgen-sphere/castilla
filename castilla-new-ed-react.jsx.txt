export default function App() {
  // Global App State
  const [currentView, setCurrentView] = useState('home');
  const [params, setParams] = useState({});
  const [isLoading, setIsLoading] = useState(true);
  
  // Data State
  const [products, setProducts] = useState(initialProducts);
  const [categories, setCategories] = useState(initialCategories);
  const [cart, setCart] = useState([]);
  const [wishlist, setWishlist] = useState([]);
  const [orders, setOrders] = useState([]);
  
  // App UI State
  const [shopFilter, setShopFilter] = useState('All');
  const [shopSort, setShopSort] = useState('Default');
  const [discountCode, setDiscountCode] = useState(null);
  const [toasts, setToasts] = useState([]);
  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  const [exitPopupShown, setExitPopupShown] = useState(false);

  // Admin State
  const [adminLoggedIn, setAdminLoggedIn] = useState(false);
  const [adminTab, setAdminTab] = useState('dashboard');

  // Modals State
  const [modals, setModals] = useState({
    search: false,
    quickView: null,
    wishlist: false,
    order: null,
    adminProduct: null,
    isAdminProductOpen: false,
    exitIntent: false
  });

  // Handle Hash Routing
  useEffect(() => {
    const handleHashChange = () => {
      let hash = window.location.hash.replace('#', '') || 'home';
      let parsedParams = {};
      
      if(hash.includes('?')) {
        const [path, qs] = hash.split('?');
        hash = path;
        const urlParams = new URLSearchParams(qs);
        for(let [k, v] of urlParams.entries()) {
          parsedParams[k] = v;
        }
      }

      if (hash === 'shop' && parsedParams.category) {
        setShopFilter(parsedParams.category);
      } else if (hash === 'shop' && !parsedParams.category) {
        setShopFilter('All');
      }

      setCurrentView(hash);
      setParams(parsedParams);
      window.scrollTo(0, 0);
      setIsMobileMenuOpen(false);
      setModals(prev => ({ ...prev, search: false }));
    };

    window.addEventListener('hashchange', handleHashChange);
    handleHashChange(); // Init
    return () => window.removeEventListener('hashchange', handleHashChange);
  }, []);

  // Loading Screen Effect
  useEffect(() => {
    const timer = setTimeout(() => setIsLoading(false), 2500);
    return () => clearTimeout(timer);
  }, []);

  // Exit Intent
  useEffect(() => {
    const handleMouseLeave = (e) => {
      if (e.clientY < 0 && !exitPopupShown) {
        setExitPopupShown(true);
        openModal('exitIntent');
      }
    };
    document.addEventListener('mouseleave', handleMouseLeave);
    return () => document.removeEventListener('mouseleave', handleMouseLeave);
  }, [exitPopupShown]);

  // Global Helpers
  const showToast = (message, type = 'success') => {
    const id = Date.now() + Math.random();
    setToasts(prev => [...prev, { id, message, type }]);
    setTimeout(() => {
      setToasts(prev => prev.filter(t => t.id !== id));
    }, 3000);
  };

  const openModal = (modalName, data = true) => {
    setModals(prev => ({ ...prev, [modalName]: data }));
  };

  const closeModal = (modalName) => {
    setModals(prev => ({ ...prev, [modalName]: false }));
  };

  // Cart & Wishlist Logic
  const toggleWishlist = (productId) => {
    setWishlist(prev => {
      if (prev.includes(productId)) {
        showToast('Removed from wishlist');
        return prev.filter(id => id !== productId);
      } else {
        showToast('Added to wishlist ❤️');
        return [...prev, productId];
      }
    });
  };

  const addToCart = (product, quantity = 1, options = null) => {
    setCart(prev => {
      if (options) {
        showToast('Custom Gift Pack added to cart!');
        return [...prev, { ...product, cartId: Date.now() + Math.random(), quantity, options }];
      }
      
      const existing = prev.find(i => i.id === product.id && !i.options);
      showToast(`${product.name} added to cart!`);
      if (existing) {
        return prev.map(i => i.cartId === existing.cartId ? { ...i, quantity: i.quantity + quantity } : i);
      }
      return [...prev, { ...product, cartId: Date.now() + Math.random(), quantity }];
    });
  };

  const updateCartQty = (cartId, delta) => {
    setCart(prev => {
      return prev.map(item => {
        if (item.cartId === cartId) {
          const newQty = item.quantity + delta;
          if (newQty < 1) return null; // handled via filter below
          return { ...item, quantity: newQty };
        }
        return item;
      }).filter(Boolean);
    });
  };

  const removeFromCart = (cartId) => {
    setCart(prev => prev.filter(item => item.cartId !== cartId));
  };

  const clearCart = () => setCart([]);

  return (
    <div className="font-sans antialiased text-[#2C2C2C] bg-[#FEFDFB] min-h-screen">
      <CustomStyles />
      
      {isLoading && <LoadingScreen />}

      <AnnouncementBar />
      <Navbar 
        cartCount={cart.reduce((sum, item) => sum + item.quantity, 0)} 
        wishlistCount={wishlist.length}
        isMobileMenuOpen={isMobileMenuOpen}
        setIsMobileMenuOpen={setIsMobileMenuOpen}
        openSearch={() => openModal('search')}
        openWishlist={() => openModal('wishlist')}
      />

      <main className="min-h-[calc(100vh-300px)] pt-[100px]">
        {currentView === 'home' && <HomeView products={products} categories={categories} toggleWishlist={toggleWishlist} wishlist={wishlist} openQuickView={(id) => openModal('quickView', id)} />}
        {currentView === 'shop' && <ShopView products={products} categories={categories} shopFilter={shopFilter} setShopFilter={setShopFilter} shopSort={shopSort} setShopSort={setShopSort} toggleWishlist={toggleWishlist} wishlist={wishlist} openQuickView={(id) => openModal('quickView', id)} />}
        {currentView.startsWith('product-') && <ProductDetailView productId={parseInt(currentView.split('-')[1])} products={products} addToCart={addToCart} toggleWishlist={toggleWishlist} wishlist={wishlist} />}
        {currentView === 'checkout' && <CartCheckoutView cart={cart} updateCartQty={updateCartQty} removeFromCart={removeFromCart} discountCode={discountCode} setDiscountCode={setDiscountCode} showToast={showToast} clearCart={clearCart} setOrders={setOrders} />}
        {currentView === 'admin' && <AdminView products={products} setProducts={setProducts} categories={categories} setCategories={setCategories} orders={orders} setOrders={setOrders} adminLoggedIn={adminLoggedIn} setAdminLoggedIn={setAdminLoggedIn} adminTab={adminTab} setAdminTab={setAdminTab} showToast={showToast} openOrderModal={(id) => openModal('order', id)} openAdminProductModal={(p) => openModal('adminProduct', p)} closeAdminProductModal={() => closeModal('adminProduct')} isProductModalOpen={modals.adminProduct !== false} />}
        {currentView === 'shipping' && <ShippingView />}
        {currentView === 'returns' && <ReturnsView />}
        {currentView === 'candle-care' && <CandleCareView />}
        {currentView === 'faqs' && <FAQsView />}
      </main>

      <Footer />

      {/* Overlays & Modals */}
      <SearchOverlay isOpen={modals.search} close={() => closeModal('search')} products={products} searchQuery={searchQuery} setSearchQuery={setSearchQuery} />
      
      {modals.quickView && <QuickViewModal product={products.find(p => p.id === modals.quickView)} close={() => closeModal('quickView')} addToCart={addToCart} />}
      
      {modals.wishlist && <WishlistModal wishlistIds={wishlist} products={products} close={() => closeModal('wishlist')} toggleWishlist={toggleWishlist} addToCart={addToCart} />}
      
      {modals.order && <OrderModal order={orders.find(o => o.id === modals.order)} close={() => closeModal('order')} setOrders={setOrders} showToast={showToast} />}
      
      {modals.exitIntent && <ExitIntentModal close={() => closeModal('exitIntent')} showToast={showToast} />}
      
      {/* Toasts */}
      <div className="fixed top-[100px] right-8 z-[10005] flex flex-col gap-4">
        {toasts.map(toast => (
          <div key={toast.id} className="bg-white p-4 rounded-xl border-l-4 border-[#C9A57B] shadow-2xl min-w-[300px] flex items-center gap-4 animate-slide-in">
            <span className="text-2xl">{toast.type === 'error' ? '❌' : '✨'}</span>
            <div>
              <strong className="block font-playfair text-[#C9A57B]">{toast.type === 'error' ? 'Oops!' : 'Success'}</strong>
              <span className="text-sm">{toast.message}</span>
            </div>
          </div>
        ))}
      </div>

    </div>
  );
}

const SearchOverlay = ({ isOpen, close, products, searchQuery, setSearchQuery }) => {
  if (!isOpen) return null;

  const matches = searchQuery.length > 1 
    ? products.filter(p => 
        p.name.toLowerCase().includes(searchQuery.toLowerCase()) || 
        p.category.toLowerCase().includes(searchQuery.toLowerCase())
      ).slice(0, 5)
    : [];

  return (
    <div className="fixed inset-0 bg-white/98 z-[9999] flex flex-col items-center pt-[15vh] px-4 animate-fade-in">
      <div className="w-full max-w-[700px] relative">
        <button onClick={close} className="absolute right-4 top-1/2 -translate-y-1/2 text-gray-400 hover:text-black"><X size={28}/></button>
        <input 
          type="text" 
          autoFocus 
          placeholder="Search for candles, trays, gifts..." 
          className="w-full py-6 pr-16 pl-8 text-2xl border-2 border-[#C9A57B] rounded-full shadow-lg outline-none bg-transparent"
          value={searchQuery}
          onChange={(e) => setSearchQuery(e.target.value)}
        />
        
        {searchQuery.length > 1 && (
          <div className="mt-8 bg-white rounded-2xl shadow-xl overflow-hidden animate-slide-up">
            {matches.length === 0 ? (
              <div className="p-8 text-center text-gray-500">No products found.</div>
            ) : (
              matches.map(p => (
                <a 
                  key={p.id} 
                  href={`#product-${p.id}`} 
                  onClick={close}
                  className="flex items-center p-4 border-b border-gray-100 hover:bg-gray-50 transition-colors"
                >
                  <img src={p.image} alt={p.name} className="w-16 h-16 rounded-lg object-cover mr-4" />
                  <div>
                    <div className="font-semibold">{p.name}</div>
                    <div className="text-sm text-gray-500">{p.category}</div>
                    <div className="text-[#C9A57B] font-semibold mt-1">{formatMoney(p.price)}</div>
                  </div>
                </a>
              ))
            )}
          </div>
        )}
      </div>
    </div>
  );
};

const ShopView = ({ products, categories, shopFilter, setShopFilter, shopSort, setShopSort, toggleWishlist, wishlist, openQuickView }) => {
  const filtered = useMemo(() => {
    let f = shopFilter === 'All' ? [...products] : products.filter(p => p.category === shopFilter);
    if(shopSort === 'Price: Low to High') f.sort((a,b) => a.price - b.price);
    if(shopSort === 'Price: High to Low') f.sort((a,b) => b.price - a.price);
    if(shopSort === 'Name: A-Z') f.sort((a,b) => a.name.localeCompare(b.name));
    if(shopSort === 'Name: Z-A') f.sort((a,b) => b.name.localeCompare(a.name));
    if(shopSort === 'Newest First') f.sort((a,b) => (b.badge==='New'?1:0) - (a.badge==='New'?1:0));
    return f;
  }, [products, shopFilter, shopSort]);

  return (
    <div className="max-w-[1280px] mx-auto px-8 animate-fade-in-view pb-24">
      <a href="#home" className="text-gray-400 hover:text-[#C9A57B] mb-8 inline-block">&larr; Back to Home</a>
      <div className="flex flex-col md:flex-row justify-between items-end mb-8 gap-4">
        <h1 className="font-playfair text-5xl">Our Collection</h1>
        <select className="border border-[#C9A57B] rounded-lg px-4 py-2 bg-white text-gray-700 outline-none focus:ring-2 focus:ring-[#C9A57B]/30" value={shopSort} onChange={(e) => setShopSort(e.target.value)}>
          <option value="Default">Sort: Default</option>
          <option value="Price: Low to High">Price: Low to High</option>
          <option value="Price: High to Low">Price: High to Low</option>
          <option value="Name: A-Z">Name: A-Z</option>
          <option value="Newest First">Newest First</option>
        </select>
      </div>

      <div className="flex flex-wrap gap-3 mb-12">
        {['All', ...categories.map(c=>c.name)].map(name => (
          <button key={name} onClick={() => setShopFilter(name)} className={`px-6 py-2 rounded-full border border-[#C9A57B] transition-colors ${shopFilter === name ? 'bg-[#C9A57B] text-white' : 'bg-transparent text-gray-700 hover:bg-[#C9A57B] hover:text-white'}`}>
            {name}
          </button>
        ))}
      </div>

      <div className="text-gray-500 mb-6">Showing {filtered.length} products</div>
      
      {filtered.length > 0 ? (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-10">
          {filtered.map(p => <ProductCard key={p.id} product={p} inWishlist={wishlist.includes(p.id)} toggleWishlist={toggleWishlist} openQuickView={openQuickView} />)}
        </div>
      ) : (
        <div className="text-center py-24 text-gray-400 text-xl">No products found in this category.</div>
      )}
    </div>
  );
};

const ProductDetailView = ({ productId, products, addToCart, toggleWishlist, wishlist }) => {
  const product = products.find(p => p.id === productId);
  const [qty, setQty] = useState(1);
  
  // Custom Gift Builder State
  const [selectedBuilderItems, setSelectedBuilderItems] = useState([]);
  const [selectedWrap, setSelectedWrap] = useState('standard');
  const [giftMessage, setGiftMessage] = useState('');

  if(!product) return <div className="p-24 text-center">Product not found.</div>;

  const inWishlist = wishlist.includes(product.id);

  const toggleBuilderItem = (id) => {
    setSelectedBuilderItems(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]);
  };

  const calculateCustomTotal = () => {
    let sum = 0;
    selectedBuilderItems.forEach(id => {
      const item = BUILDER_ITEMS.find(i => i.id === id);
      if(item) sum += item.price;
    });
    const wrapOption = WRAP_OPTIONS.find(w => w.id === selectedWrap);
    if(wrapOption) sum += wrapOption.price;
    return sum;
  };

  const handleCustomAdd = () => {
    const itemNames = selectedBuilderItems.map(id => BUILDER_ITEMS.find(i => i.id === id)?.name);
    const wrapName = WRAP_OPTIONS.find(w => w.id === selectedWrap)?.name;
    
    addToCart(product, 1, { 
      price: calculateCustomTotal(), 
      desc: itemNames.length ? `${itemNames.join(', ')} | ${wrapName}` : `Empty ${wrapName}`, 
      msg: giftMessage 
    });
  };

  return (
    <div className="max-w-[1280px] mx-auto px-8 animate-fade-in-view pb-24">
      <a href="#shop" className="text-gray-400 hover:text-[#C9A57B] mb-8 inline-block">&larr; Back to Shop</a>
      
      <div className="grid grid-cols-1 md:grid-cols-2 gap-16 items-start">
        <img src={product.image} alt={product.name} className="w-full aspect-[4/5] object-cover rounded-3xl border border-[#C9A57B]/20 shadow-lg sticky top-[120px]" />
        
        <div>
          <div className="text-[#C9A57B] uppercase tracking-widest text-sm font-bold mb-2">{product.category}</div>

            <div className="bg-[#F9F7F4] p-6 rounded-xl mb-8 space-y-2">
              <div><strong className="text-[#C9A57B]">Top Notes:</strong> {product.topNotes}</div>
              <div><strong className="text-[#C9A57B]">Heart Notes:</strong> {product.heartNotes}</div>
              <div><strong className="text-[#C9A57B]">Base Notes:</strong> {product.baseNotes}</div>

            {product.burnTime && <span>🕯️ Burn Time: ~{product.burnTime} hours</span>}

            <span>🇵🇰 Handcrafted in Pakistan</span>
          </div>

          {product.category === 'Custom Gift Packs' ? (
            <div className="border border-gray-200 rounded-2xl p-8 mb-8 bg-white shadow-sm">
              <h3 className="font-playfair text-2xl mb-6 text-[#C9A57B]">1. Curate Your Box</h3>
              <div className="grid grid-cols-2 gap-4 mb-10">
                {BUILDER_ITEMS.map(item => (
                  <div 
                    key={item.id} 
                    onClick={() => toggleBuilderItem(item.id)}
                    className={`cursor-pointer rounded-xl p-4 flex flex-col items-center text-center transition-all border-2 ${selectedBuilderItems.includes(item.id) ? 'border-[#C9A57B] bg-[#C9A57B]/5' : 'border-gray-100 hover:border-[#C9A57B]/50'}`}

const CartCheckoutView = ({ cart, updateCartQty, removeFromCart, discountCode, setDiscountCode, showToast, clearCart, setOrders }) => {
  const [step, setStep] = useState('cart'); // cart | checkout
  const [formData, setFormData] = useState({ fname: '', lname: '', phone: '', address: '', city: '', payment: 'cod' });
  
  const subtotal = cart.reduce((sum, item) => sum + ((item.options ? item.options.price : item.price) * item.quantity), 0);
  const discountAmount = discountCode === 'WELCOME15' ? subtotal * 0.15 : 0;
  const discountedSubtotal = subtotal - discountAmount;
  const delivery = (discountedSubtotal >= 5000 || discountedSubtotal === 0) ? 0 : 300;
  const total = discountedSubtotal + delivery;

  const [promoInput, setPromoInput] = useState(discountCode || '');

  const handleApplyPromo = () => {
    const val = promoInput.toUpperCase().trim();
    if (val === 'WELCOME15') {
      setDiscountCode(val);
      showToast('Voucher applied! 15% off.', 'success');
    } else if (val === '') {
      setDiscountCode(null);
      showToast('Voucher removed.', 'success');
    } else {
      setDiscountCode(null);
      showToast('Invalid voucher code.', 'error');
    }
  };

  const handlePlaceOrder = (e) => {
    e.preventDefault();
    const orderId = 'ORD-' + Math.floor(1000 + Math.random() * 9000);
    setOrders(prev => [...prev, {
      id: orderId,
      customer: { name: `${formData.fname} ${formData.lname}`, phone: formData.phone, address: formData.address, city: formData.city },
      total,
      status: 'Pending',
      date: new Date().toLocaleDateString(),
      items: [...cart]
    }]);
    clearCart();
    setDiscountCode(null);
    setStep('success');

      <div className="grid grid-cols-1 lg:grid-cols-5 gap-10 items-start">
        
        {/* Left Column */}
        <div className="lg:col-span-3">

const AdminProductModal = ({ categories, products, setProducts, close, showToast }) => {
  // Using a local trick to find which product to edit if the window has a stored ID
  // In a full refactor we would pass it down, but since we are replacing the global method:
  const editId = window.adminEditId; // Hack to bridge state easily
  const p = editId ? products.find(x => x.id === editId) : null;
  
  const [form, setForm] = useState({
    name: p ? p.name : '',
    category: p ? p.category : categories[0].name,
    price: p ? p.price : '',
    originalPrice: p && p.originalPrice ? p.originalPrice : '',
    image: p ? p.image : '',
    description: p ? p.description : '',
    scentStory: p ? p.scentStory : '',
    badge: p && p.badge ? p.badge : '',
    burnTime: p && p.burnTime ? p.burnTime : '',
    weight: p && p.weight ? p.weight : ''
  });

  const handleSubmit = (e) => {
    e.preventDefault();
    const prodData = {
      id: p ? p.id : Date.now(),
      name: form.name,
      category: form.category,
      price: parseInt(form.price) || 0,
      originalPrice: parseInt(form.originalPrice) || null,
      image: form.image,
      description: form.description,
      scentStory: form.scentStory || 'A beautifully crafted piece by Castilla.',
      topNotes: p ? p.topNotes : 'N/A',
      heartNotes: p ? p.heartNotes : 'N/A',
      baseNotes: p ? p.baseNotes : 'N/A',
      badge: form.badge || null,
      burnTime: parseInt(form.burnTime) || null,
      weight: parseInt(form.weight) || null
    };

    if (p) {
      setProducts(products.map(x => x.id === p.id ? prodData : x));
      showToast('Product updated successfully!');
    } else {
      setProducts([...products, prodData]);
      showToast('Product created successfully!');
    }
    close();
  };

  return (
    <div className="fixed inset-0 bg-black/60 backdrop-blur-sm z-[10002] flex items-center justify-center p-4 animate-fade-in" onClick={close}>
      <div className="bg-white rounded-3xl w-full max-w-[700px] max-h-[90vh] overflow-y-auto relative p-10 animate-slide-up" onClick={e => e.stopPropagation()}>
        <button onClick={close} className="absolute top-6 right-6 w-10 h-10 bg-white rounded-full shadow-md flex items-center justify-center z-10 hover:text-[#C9A57B]"><X size={20}/></button>
        <h2 className="font-playfair text-3xl mb-8">{p ? 'Edit Product' : 'Add New Product'}</h2>
        
        <form onSubmit={handleSubmit} className="space-y-6">
          <div className="grid grid-cols-2 gap-4">
            <div><label className="block mb-2 font-medium">Name *</label><input required className="input-field" value={form.name} onChange={e=>setForm({...form, name: e.target.value})} /></div>
            <div>
              <label className="block mb-2 font-medium">Category *</label>
              <select required className="input-field" value={form.category} onChange={e=>setForm({...form, category: e.target.value})}>
                {categories.map(c => <option key={c.name} value={c.name}>{c.name}</option>)}
              </select>
            </div>
          </div>
          
          <div className="grid grid-cols-2 gap-4">
            <div><label className="block mb-2 font-medium">Price (PKR) *</label><input type="number" required className="input-field" value={form.price} onChange={e=>setForm({...form, price: e.target.value})} /></div>
            <div><label className="block mb-2 font-medium">Original Price</label><input type="number" className="input-field" value={form.originalPrice} onChange={e=>setForm({...form, originalPrice: e.target.value})} /></div>
          </div>
          
          <div><label className="block mb-2 font-medium">Image URL *</label><input type="url" required className="input-field" value={form.image} onChange={e=>setForm({...form, image: e.target.value})} /></div>
          
          <div><label className="block mb-2 font-medium">Short Description *</label><input type="text" required className="input-field" value={form.description} onChange={e=>setForm({...form, description: e.target.value})} /></div>
          
          <div><label className="block mb-2 font-medium">Scent Story</label><textarea rows="2" className="input-field" value={form.scentStory} onChange={e=>setForm({...form, scentStory: e.target.value})} /></div>
          
          <div className="grid grid-cols-3 gap-4">
            <div>
              <label className="block mb-2 font-medium">Badge</label>
              <select className="input-field" value={form.badge} onChange={e=>setForm({...form, badge: e.target.value})}>
                <option value="">None</option>
                <option value="New">New</option>
                <option value="Bestseller">Bestseller</option>
                <option value="Sale">Sale</option>
                <option value="✨">✨</option>
              </select>
            </div>
            <div><label className="block mb-2 font-medium">Burn Time (hrs)</label><input type="number" className="input-field" value={form.burnTime} onChange={e=>setForm({...form, burnTime: e.target.value})} /></div>
            <div><label className="block mb-2 font-medium">Weight (g)</label><input type="number" className="input-field" value={form.weight} onChange={e=>setForm({...form, weight: e.target.value})} /></div>
